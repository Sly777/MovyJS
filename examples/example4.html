Custom Move Example
    $(document).ready(function (){
        $('.mouseArea').movyjs({
            moveType: "custom",
            customMoveType: ["left", "top", "right", "bottom"],
            onEveryTick: function (count){
                console.log(count + ".move completed.");
            },
            onComplete: function (){
                alert("custom move completed");
            }
        });
    });
